QHTML7 Behavior Animation Loop Test

This page loops a property animation from 1vh to 100vh and back using behavior on and q-script-action.

q-component behavior-loop-bar { q-property barWidth: 1 q-property nextBarWidth: 100 behavior on barWidth { q-sequential-animation { q-property-animation { duration: 1400 steps: 80 onstepped(stepNum, value) { this.ownerElement.querySelector(".behavior-loop-fill").style.width = String(value) + "vh"; this.ownerElement.querySelector(".behavior-loop-readout").textContent = "step " + String(stepNum) + ": " + String(Math.round(value)) + "vh"; } } q-script-action { this.nextBarWidth = this.barWidth == 100 ? 1 : 100; setTimeout(function() { this.barWidth = this.nextBarWidth; }.bind(this), 0); } } } q-style testShellStyle { padding: 16px; borderWidth: 1px; borderColor: #cbd5e1; borderRadius: 8px; backgroundColor: #f8fafc; } q-style testTrackStyle { width: 100%; height: 34px; padding: 4px; borderWidth: 1px; borderColor: #d8e0eb; borderRadius: 8px; backgroundColor: #ffffff; } q-style testBarStyle { width: 1vh; height: 24px; borderRadius: 6px; backgroundColor: #22c55e; } q-style testReadoutStyle { marginTop: 10px; color: #475569; fontSize: 13px; fontWeight: 700; } q-default-theme behaviorLoopTheme { .behavior-loop-shell { testShellStyle } .behavior-loop-track { testTrackStyle } .behavior-loop-fill { testBarStyle } .behavior-loop-readout { testReadoutStyle } } onready { this.querySelector(".behavior-loop-fill").style.width = String(this.barWidth) + "vh"; this.querySelector(".behavior-loop-readout").textContent = "width: " + String(this.barWidth) + "vh"; this.barWidth = 100; } onbarWidthchanged(value) { this.querySelector(".behavior-loop-fill").style.width = String(value) + "vh"; this.querySelector(".behavior-loop-readout").textContent = "width: " + String(value) + "vh"; } div.behavior-loop-shell { div.behavior-loop-track { div.behavior-loop-fill { } } div.behavior-loop-readout { text { width: 1vh } } } } behavior-loop-bar loopDemo { }
